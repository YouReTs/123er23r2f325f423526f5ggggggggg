# Определение кодировки одного файла .txt
# https://ru.stackoverflow.com/questions/700664/%D0%A3%D0%B7%D0%BD%D0%B0%D1%82%D1%8C-%D0%BA%D0%BE%D0%B4%D0%B8%D1%80%D0%BE%D0%B2%D0%BA%D1%83-%D1%84%D0%B0%D0%B9%D0%BB%D0%B0

from chardet.universaldetector import UniversalDetector

detector = UniversalDetector()
with open('text.txt', 'rb') as fh:
    for line in fh:
        detector.feed(line)
        if detector.done:
            break
    detector.close()
print(detector.result)
